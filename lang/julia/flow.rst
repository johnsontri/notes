for
----------------------------------------------------------------------

::

    for i = [1, 2, 3]
        println(i)
    end

    for i in [1, 2, 3]
        println(i)
    end

    for i in 1:5
        println(i^2)
    end

::

    for i in Dict("foo" => 1, "bar" => 2)
        println(i)
    end

    for (k, v) in Dict("foo" => 1, "bar" => 2)
        println(k, ": ", v)
    end
